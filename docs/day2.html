choice = int(input("Enter choice: "))

if choice == 1:
    r = float(input("Enter radius: "))
    area = 3.14 * r * r
    print("Area of circle:", area)

elif choice == 2:
    l = float(input("Enter length: "))
    w = float(input("Enter width: "))
    area = l * w
    print("Area of rectangle:", area)

else:
    print("Invalid choice")
