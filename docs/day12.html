    for i in range(n):
        print(a, end=" ")

def is_armstrong(num):
    temp = num
    power = len(str(num))
    total = 0

    while temp > 0:
        digit = temp % 10
        total += digit ** power
        temp //= 10

    return total == num

n = int(input("Enter number: "))

if is_armstrong(n):
    print("Armstrong Number")
else:
    print("Not Armstrong")

def is_perfect(n):
    total = 0

    for i in range(1, n):
        if n % i == 0:
            total += i

    return total == n

num = int(input("Enter number: "))

if is_perfect(num):
    print("Perfect Number")
else:
    print("Not Perfect")
