nums = sorted(list(map(int, input("Enter sorted numbers: ").split())))
key = int(input("Enter element to search: "))

low = 0
high = len(nums) - 1

while low <= high:
    mid = (low + high) // 2

    if nums[mid] == key:
        print("Found at index:", mid)
        break
    elif nums[mid] < key:
        low = mid + 1
    else:
        high = mid - 1
else:
    print("Not found")
