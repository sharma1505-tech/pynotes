class Student:
    def __init__(self, name, marks):
        self.name = name
        self.marks = marks

    def grade(self):
        if self.marks >= 90:
            print(self.name, "- Grade A")
        elif self.marks >= 50:
            print(self.name, "- Pass")
        else:
            print(self.name, "- Fail")

name = input("Enter name: ")
marks = int(input("Enter marks: "))

s = Student(name, marks)
s.grade()

Enter marks:  91
